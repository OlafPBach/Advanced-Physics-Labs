const int ledPin = 9;
const int Pi = 3.1415926;

int brightness;

void setup() {
Serial.begin(500000);
pinMode(ledPin, OUTPUT);
  }

void loop() {
brightness = 100+50*sin(2*Pi*millis()/1000);
analogWrite(ledPin,brightness);
int sensorValue = analogRead(A0);
long Time = millis();
Serial.print(Time);
Serial.print(',');
Serial.println(sensorValue);
delay(1);
  }
